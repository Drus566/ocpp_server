add_custom_target(final_post_build ALL
   DEPENDS ocpp_server open-ocpp-dynamic
	COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_SOURCE_DIR}/app/bin
	COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_SOURCE_DIR}/app/lib
	COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_SOURCE_DIR}/app/certificates
	COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_SOURCE_DIR}/app/config
	COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_SOURCE_DIR}/app/log
	COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_SOURCE_DIR}/app/schemas
	COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_SOURCE_DIR}/app/scripts
	COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_SOURCE_DIR}/app/database

	COMMAND ${CMAKE_COMMAND} -E copy
   ${CMAKE_SOURCE_DIR}/bin/ocpp_server
   ${CMAKE_SOURCE_DIR}/app/bin/ocpp_server

	COMMAND ${CMAKE_COMMAND} -E copy
   ${CMAKE_SOURCE_DIR}/lib/libocpp_server_dynamic.so
   ${CMAKE_SOURCE_DIR}/app/lib/libocpp_server_dynamic.so

	COMMAND ${CMAKE_COMMAND} -E copy
   ${CMAKE_SOURCE_DIR}/lib/ocpp/libopen-ocpp.so.2.0.0
   ${CMAKE_SOURCE_DIR}/app/lib/libopen-ocpp.so.2.0.0

	COMMAND ${CMAKE_COMMAND} -E copy
   ${CMAKE_SOURCE_DIR}/config/ocpp_config.ini
   ${CMAKE_SOURCE_DIR}/app/config/ocpp_config.ini

	COMMAND ${CMAKE_COMMAND} -E copy_directory
   ${CMAKE_SOURCE_DIR}/3rdparty/ocpp/schemas
   ${CMAKE_SOURCE_DIR}/app/schemas

   COMMAND ${CMAKE_COMMAND} -E copy_directory
   ${CMAKE_SOURCE_DIR}/certificates
   ${CMAKE_SOURCE_DIR}/app/certificates

   COMMAND ${CMAKE_COMMAND} -E copy_directory
   ${CMAKE_SOURCE_DIR}/web
   ${CMAKE_SOURCE_DIR}/app/web

)
